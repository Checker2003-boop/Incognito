using System;

namespace Polyclinic
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int receptionTimeMinute = 10;

            Console.Write("Введите кол-во пациентов: ");
            int peopleQueue = Convert.ToInt32(Console.ReadLine());
            int queueTimeMinute = peopleQueue * receptionTimeMinute;
            
            int queueHours;
            int queueMinutes;
            int conversionFactorMinutesToHours = 60;
            

            queueHours = queueTimeMinute / conversionFactorMinutesToHours;
            queueMinutes = queueTimeMinute % conversionFactorMinutesToHours;
            Console.WriteLine($"Вы должны отстоять в очереди: " + queueHours + ' ' + "часов" + ' ' + queueMinutes + ' ' + "минут.");
        }
    }
}
