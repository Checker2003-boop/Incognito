using System;

namespace Sequence
{
    internal class Program
    {
        static void Main(string[] args)
        {
            int number = 5;
            int step = 7;
            int maxNumber = 103;

            for (int i = number; i <= maxNumber; i += step)
            {
                Console.Write(i + " ");
            }

            Console.ReadLine();
        }
    }
}
