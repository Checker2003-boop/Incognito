using System;

namespace Name_output
{
    internal class Program
    {
        static void Main(string[] args)
        {
            string inputUserName;
            char symbols;
            string midleLine;
            string frame = "";

            Console.Write("Введите имя которое нужно вставить в рамку из символов: ");
            inputUserName = Console.ReadLine();

            Console.Write("Введите какой символ использовать в качестве рамки: ");
            symbols = (char)Console.Read();

            midleLine = $"{symbols} {inputUserName} {symbols}";

            for (int i = 0; i < midleLine.Length; ++i)
            {
                frame += symbols;
            }

            Console.WriteLine($"{frame}\n{midleLine}\n{frame}");
        }
    }
}
